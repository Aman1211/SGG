Open DataTable

Default column sort

Required files
jquery-2.2.4.js
simple_php_datasource.php
style.css
OpenDataTable.js
country_city.sql

Code
Simply add default_sort:{sort_col:'1',sort_type:'DESC'} to enable default column sort, use 0 for col 1, 1 for col 2 and on... 
    
 <html>
   <head>
     <title>Open Data Table</title>
     <script src="https://code.jquery.com/jquery-2.2.4.js"></script>
     <link rel="stylesheet" type="text/css" href="style.css">
     <script src="OpenDataTable.js"></script>
     <script type="text/javascript">
    $(document).ready(function(){
      $(".OpenDataTable").OpenDataTable({  
        url:"simple_php_datasource.php",
        default_sort:{sort_col:'1',sort_type:'DESC'}, // use 0 for col 1, 1 for col 2  and on...
      });
    });
     </script>
   </head>
   <body>
     <table class="OpenDataTable">
	       <thead>
		       <tr>
			        <th>Code</th>
			        <th>Name</th>
			        <th>Continent</th>
		        </tr>
	       </thead>
       <tbody></tbody>
     </table>
   </body>
 </html>
